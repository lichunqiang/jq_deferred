            var wait = function() {
              var dtd = $.Deferred();
              var task = function() {
                alert('Look at console of reject');
                dtd.reject('i am resolved')
              }

              setTimeout(task, 3e3);

              return dtd.promise(); //deferred的高级封装只暴露了then等方法,没有resolve,reject等方法
            }

            wait().done(function(msg) {
              console.log('sucess......', msg, 'end sucess...')
            })
            .fail(function(msg) {
              console.log('fail......')
              console.log('this:', this);
              console.log('msg:', msg);
              console.log('end fail......')
            });
          
        
          
            var wait = function() {
              var dtd = $.Deferred();
              var task = function() {
                alert('Look at console of rejectWith');
                dtd.rejectWith({id: 1}, ['i am resolved'])
              }

              setTimeout(task, 3e3);

              return dtd.promise(); //deferred的高级封装只暴露了then等方法,没有resolve,reject等方法
            }

            wait().done(function(msg) {
              console.log('sucessWith......', msg, 'end sucessWith...')
            })
            .fail(function(msg) {
              console.log('');
              console.log('failWith......')
              console.log('this:', this);
              console.log('msg:', msg);
              console.log('end failWith......')
            });
